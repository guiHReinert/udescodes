filter::(a->Bool) -> [a] -> [a]

filter p [] = []
filter p (x:xs) = if p x then x:filter p xs else filter p xs

filter'' p (x:xs) = [x|x <- xs, p x]

fold'::(a->b->c) -> b -> [a] -> b

fold' f e [] = e 
fold' f e (x:xs) = f x (fold' f e xs)
